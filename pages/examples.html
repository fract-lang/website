Simple Calculator
print("Welcome to Fract simple calculator!") print("**** Operations ****") print("1 - Add") print("2 - Subtract") print("3 - Multiple") print("4 - Divide") print("0 - Exit") for const input = input("Select your operation: ") if input == "0" break elif input == "1" || input == "2" || input == "3" || input == "4" const ( first = float(input("Enter first number: ")) second = float(input("Enter second number: ")) ) if input == "1" print(first + second) elif input == "2" print(first - second) elif input == "3" print(first * second) else print(first / second) end else print("Invalid choice!") end end

Fibonacci
var ( nterms = int(input("How many terms: ")) n1 = 0 n2 = 1 count = 0 ) if nterms <= 0 print("Please enter a positive integer!") else print("Fibonacci sequence;") for count < nterms print(n1) const nth = n1 + n2 # update values n1 = n2 n2 = nth count += 1 end end

FizzBuzz
for _, number in range(1, 100) if number % 3 == 0 && number % 5 == 0 print('FizzBuzz') elif number % 3 == 0 print('Fizz') elif number % 5 == 0 print('Buzz') else print(number) end end

Prime Number
for _, x in range(2, 100) if x == 2 print(x) continue elif x < 2 || x % 2 == 0 continue end var prime = true for _, y in range(2, x - 1) if x % y == 0 prime = false break end end if prime print(x) end end
