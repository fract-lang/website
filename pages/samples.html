Simple Calculator
package main print('Welcome to Fract simple calculator!') println('**** Operations ****') println('1 - Add') println('2 - Subtract') println('3 - Multiple') println('4 - Divide') println('0 - Exit') while { choice = input('Select your operation: ') if choice == '0' { break } else if choice in ['1', '2', '3', '4'] { first = float(input('Enter first number: ')) second = float(input('Enter second number: ')) if choice == '1' { println(first + second) } else if choice == '2' { println(first - second) } else if choice == '3' { println(first * second) } else { println(first / second) } } else { println('Invalid choice!') } }


Fibonacci
package main nterms = int(input('How many terms: ')) n1 = 0 n2 = 1 count = 0 if nterms <= 0 { println('Please enter a positive integer!') } else { println('Fibonacci sequence;') while count < nterms { println(n1) nth = n1 + n2 # update values n1 = n2 n2 = nth count++ } }


FizzBuzz
package main foreach _, number in range(1, 100) { if number % 3 == 0 && number % 5 == 0 { println('FizzBuzz') } else if number % 3 == 0 { println('Fizz') } else if number % 5 == 0 { println('Buzz') } else { println(number) } }


Prime Number
package main foreach _, x in range(2, 100) { if x == 2 { println(x) continue } else if x < 2 || x % 2 == 0 { continue } prime = true foreach _, y in range(2, x - 1) { if x % y == 0 { prime = false break } } if prime { println(x) } }
